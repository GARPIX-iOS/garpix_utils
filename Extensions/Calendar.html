  • periodBetweenDays(_:_:with:)

    Функция возвращает количество дней, месяцев или лет между двумя датами

      let startDate = Date()
  let endDate = Date()
  days = Calendar.current.periodBetweenDays(startDate, endDate, with: [.day])
  // output -> количество дней (0)
